    def resizeEvent(self, event:QtGui.QResizeEvent):
        super(TopUi, self).resizeEvent(QtGui.QResizeEvent)
        self.title.setFixedWidth(self.parent().width())

    def mousePressEvent(self, event):
        self.start = self.mapToGlobal(event.pos())
        self.pressing = True

    def mouseMoveEvent(self, event):
        if self.pressing:
            self.end = self.mapToGlobal(event.pos())
            self.movement = self.end-self.start
            self.parent().setGeometry(self.mapToGlobal(self.movement).x(),
                                    self.mapToGlobal(self.movement).y(),
                                    self.parent().width(),
                                    self.parent().height())
            self.start = self.end

    def mouseReleaseEvent(self, event:PySide2.QtGui.QMouseEvent):
        self.pressing = False

    def btn_close_clicked(self):
        self.parent().close()

    def btn_max_clicked(self):
        self.parent().showMaximized()

    def btn_min_clicked(self):
        self.parent().showMinimized()



        btn_minimize = QPushButton()                            # Minimize
        btn_minimize.setObjectName("btn_minimize")
        btn_minimize.setSizePolicy(sizePolicy)
        btn_minimize.clicked.connect(lambda: self.parent().showMinimized())

        btn_Maximize = QPushButton()                            # Maximise
        btn_Maximize.setObjectName("btn_Maximize")
        btn_Maximize.setSizePolicy(sizePolicy)
        btn_Maximize.clicked.connect(self.btn_max_clicked)

        btn_close = QPushButton()                               # Close
        btn_close.setObjectName("btn_close")
        btn_close.setSizePolicy(sizePolicy)
        btn_close.clicked.connect(self.btn_close_clicked)

        self.title = QLabel("Assistant graphique pour SAT")
        self.title.setFixedHeight(35)

        hbox_title.addWidget(self.title)
        hbox_title.addWidget(btn_minimize)                      # Rassemblement des boutons
        hbox_title.addWidget(btn_Maximize)
        hbox_title.addWidget(btn_close)
